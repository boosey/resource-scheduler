
# Build for OCP and deploy 
./mvnw clean package -Dquarkus.container-image.build=true -Dquarkus.kubernetes.deploy=true

# Create deployment config
kubectl create -n resource-scheduler -f src/main/k8s/ReservationCommandGrpc.yaml

# Build locally and deploy
mvn clean install \
&& docker build -f src/main/docker/Dockerfile.jvm -t docker.io/boosey/reservations-command-grpc . \
&& docker push docker.io/boosey/reservations-command-grpc \
&& oc rollout latest deploymentconfig/reservations-command-grpc
